%% Let's test with two options on a book citation
%% 1.9.14 omits the e.g. and puts in the tilde
\verb#\citep[e.g.][chap.~2]{bustos1999}#\\
\citep[e.g.][chap.~2]{bustos1999} \\
(e.g. Bustos et al., 1999, chap. 2)\\
\hline
%% Let's test one option on a book citation
%% According to p. 7 of  Daly's natbib guide, a single
%% option is a post--note.  1.9.14 makes it a pre--note and omits comma
\verb#\citet[chap.~5]{bustos1999}#\\
\citet[chap.~5]{bustos1999}\\
Bustos et al. (1999, chap. 5)\\
 \hline 
%% Let's test with a single option on a two--author paper
%% 1.9.14 omits e.g. and puts comma-space after year
\verb#\citep[e.g.][]{chung2002}#\\
\citep[e.g.][]{chung2002} \\
(e.g. Chung and Worster, 2002)\\
 \hline
%% Try the starred versions
\verb#\citet*{chung2002, biot1941}#\\
\citet*{chung2002, biot1941}\\
Chung and Worster (2002); Biot (1941)\\
\hline
\verb#\citep*{chung2002, biot1941}#\\
\citep*{chung2002, biot1941}\\
(Chung and Worster, 2002; Biot, 1941)\\
\hline
%% Let's try a parenthetical list of authors with e.g.
%% 1.9.14 omits e.g., and uses comma-space-semicolon-space
%% as separator; it should be semicolon-space.
\verb#\citep[e.g.][]{bustos1999, chung2002, biot1941}#\\
\citep[e.g.][]{bustos1999, chung2002, biot1941} \\
(e.g. Bustos et al., 1999; Chung and Worster, 2002; Biot, 1941)\\

\end{tabular}
\end{center}
